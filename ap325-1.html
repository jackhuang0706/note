Q-1-5. 二維黑白影像編碼

#include <bits/stdc++.h>
#define fastio ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
#define int long long
using namespace std;

string s;
int idx=-1;

int solve(int n){
    int sum=0;
    idx++;
    if(s[idx]=='2'){
        for(int i=0;i<4;i++){
            sum+=solve(n/4);
        }
    }
    else if(s[idx]=='1'){
        return n;
    }
    return sum;
}
signed main(){
    fastio;
    int n;
    cin >> s >> n;
    n*=n;
    int ans=solve(n);
    cout << ans << "\n";
}

P-1-7. 子集合乘積

#include <bits/stdc++.h>
#define fastio ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
#define int long long
#define mod 10009
using namespace std;

const int maxN=30;
vector<int> arr(maxN);
int n,ans=0;

void solve(int cnt,int total){
    if(cnt>=n){
        if(total%mod==1) ans++;
        return;
    }
    solve(cnt+1,(total*arr[cnt])%mod);
    solve(cnt+1,total);
    return;
}
signed main(){
    fastio;
    cin >> n;
    for(int i=0;i<n;i++) cin >> arr[i];
    solve(0,1);
    cout << ans-1 << "\n";
}

Q-1-8. 子集合的和

#include <bits/stdc++.h>
#define fastio ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
#define int long long
using namespace std;

const int maxN=30;
vector<int> arr(maxN);
int n,p,ans=0;

void solve(int cnt,int total){
    if(cnt>=n){
        if(total<=p){
            ans=max(ans,total);
        }
        return;
    }
    solve(cnt+1,total+arr[cnt]);
    solve(cnt+1,total);
    return;
}
signed main(){
    fastio;
    cin >> n >> p;
    for(int i=0;i<n;i++) cin >> arr[i];
    solve(0,0);
    cout << ans << "\n";
}
