int solve(int x,int y,int p){
    if(y==1) return x;
    if(y&1) return (x*solve(x,y-1,p))%p;
    return (solve(x,y/2,p)*solve(x,y/2,p))%p;
}

    cin >> x >> y >> p;
    x%=p;

Q-2-5. 快速計算費式數列第 n 項

#include <bits/stdc++.h>
#define fastio ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
#define int long long
#define mod 1000000007
using namespace std;

vector<vector<int>> times(vector<vector<int>> a,vector<vector<int>> b){
    vector<vector<int>> result(2,vector<int> (2));
    result[0][0]=a[0][0]*b[0][0]+a[0][1]*b[1][0];result[0][0]%=mod;
    result[0][1]=a[0][0]*b[0][1]+a[0][1]*b[1][1];result[0][1]%=mod;
    result[1][0]=a[1][0]*b[0][0]+a[1][1]*b[1][0];result[1][0]%=mod;
    result[1][1]=a[1][0]*b[0][1]+a[1][1]*b[1][1];result[1][1]%=mod;
    return result;
}
vector<vector<int>> cal(vector<vector<int>> arr,int p){
    vector<vector<int>> result={{1,0},{0,1}};
    while(p>0){
        if(p&1){
            result=times(result,arr);
        }
        arr=times(arr,arr);
        p>>=1;
    }
    return result;
}
signed main(){
    fastio;
    int n;
    while(cin >> n){
        if(n==-1){
            break;
        }
        vector<vector<int>> arr={{1,1},{1,0}};
        vector<vector<int>> ans=cal(arr,n-1);
        ans[0][0]%=mod;
        cout << ans[0][0] << "\n";
    }
}

P-2-6. Two-Number problem

#include <bits/stdc++.h>
#define fastio ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
#define int long long
using namespace std;

signed main(){
    fastio;
    int m,n,k;
    cin >> m >> n >> k;
    vector<int> a(m),b(n);
    for(int &i:a) cin >> i;
    for(int &i:b) cin >> i;
    sort(a.begin(),a.end());
    sort(b.begin(),b.end(),greater<int>());
    int idxa=0,idxb=0,ans=0;
    while(idxa<m && idxb<n){
        if(a[idxa]+b[idxb]==k){
            ans++,idxb++;
        }
        else if(a[idxa]+b[idxb]<k){
            idxa++;
        }
        else{
            idxb++;
        }
    }
    cout << ans << "\n";
}

Q-2-7. 互補團隊

#include <bits/stdc++.h>
#define fastio ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
#define int long long
using namespace std;

signed main(){
    fastio;
    int m,n;
    cin >> m >> n;
    map<char,int> ch;
    int cnt=1;
    for(int i=1;i<=m;i++){
        ch[i+64]=cnt;
        cnt<<=1;
    }
    string s;
    vector<int> arr(n,0);
    for(int i=0;i<n;i++){
        cin >> s;
        int len=s.length();
        set<char> filt;
        for(int j=0;j<len;j++){
            filt.insert(s[j]);
        }
        for(auto &vis:filt){
            arr[i]+=ch[vis];
        }
    }
    int result=pow(2,m)-1;
    sort(arr.begin(),arr.end());
    int l=0,r=n-1,ans=0;
    while(l<r){
        if(arr[l]+arr[r]==result){
            ans++,l++,r--;
        }
        else if(arr[l]+arr[r]<result){
            l++;
        }
        else{
            r--;
        }
    }
    cout << ans << "\n";
}

Q-2-8. 模逆元

#include <bits/stdc++.h>
#define fastio ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
#define int long long
using namespace std;

int inv(int a,int n){
    return a==1?1:inv(n%a,n)*(n-n/a)%n;
}
signed main(){
    fastio;
    int n,p,x;
    cin >> n >> p;
    for(int i=0;i<n;i++){
        cin >> x;
        int ans=inv(x,p);
        cout << ans << " ";
    }
}
